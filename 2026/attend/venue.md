## Conference Venue: The Nicolaus Hotel

The conference will be organized at **The Nicolaus Hotel**, an elegant conference center and hotel located in the Poggiofranco district of Bari, Puglia, Italy.

The Nicolaus Hotel offers a welcoming atmosphere, bright interiors, and functional spaces for conferences and meetings. The venue has a large Congress Centre divided into modular and technical conference rooms.

    map.on('click',clearActive);
 
    L.marker([VENUE.lat,VENUE.lng],{icon:L.divIcon({className:'',html:'<div class="dg-venue">V</div>',iconSize:[32,32],iconAnchor:[16,16]}),zIndexOffset:1000})
     .addTo(map).bindPopup('<div class="dg-pop-name">'+VENUE.name+'</div><div class="dg-pop-meta">ISMAR 2026 venue - your starting point</div>');
 
    var markers={};
    PLACES.forEach(function(p,i){
      var mk=L.circleMarker([p.lat,p.lng],{radius:8,color:'#fff',weight:2,fillColor:CATS[p.cat].color,fillOpacity:1}).addTo(map);
      mk.bindPopup('<div class="dg-pop-name">'+p.name+'</div><div class="dg-pop-meta">'+p.cat+' · '+distLabel(p.dist)+' ('+p.walk+' min walk) · ★ '+p.rating.toFixed(1)+' · '+euro(p.price)+'<br>'+p.closed+'</div><a class="dg-pop-link" href="'+p.url+'" target="_blank" rel="noopener">Open in Google Maps ↗</a>');
      mk.on('click',function(){setActive(i,false);});
      markers[i]=mk;
    });
 
    var ALL=Object.keys(CATS);
    var state={active:new Set(ALL),sort:'distance',selected:null};
    var filtersEl=document.getElementById('dg-filters'),listEl=document.getElementById('dg-list'),countEl=document.getElementById('dg-count');
 
    var allChip=document.createElement('button');
    allChip.className='dg-chip';allChip.textContent='All';allChip.setAttribute('aria-pressed','true');
    allChip.onclick=function(){state.active=new Set(ALL);syncChips();render();};
    filtersEl.appendChild(allChip);
    ALL.forEach(function(cat){
      var b=document.createElement('button');b.className='dg-chip';b.setAttribute('aria-pressed','true');b.setAttribute('data-cat',cat);
      b.innerHTML='<span class="dg-dot" style="background:'+CATS[cat].color+'"></span>'+cat;
      b.onclick=function(){
        if(state.active.size===ALL.length){state.active=new Set([cat]);}
        else if(state.active.has(cat)){state.active.delete(cat);if(!state.active.size)state.active=new Set(ALL);}
        else{state.active.add(cat);}
        syncChips();render();
      };
      filtersEl.appendChild(b);
    });
    function syncChips(){
      var all=state.active.size===ALL.length;
      allChip.setAttribute('aria-pressed',all?'true':'false');
      filtersEl.querySelectorAll('[data-cat]').forEach(function(b){
        b.setAttribute('aria-pressed',(all||state.active.has(b.getAttribute('data-cat')))?'true':'false');
      });
    }
    document.getElementById('dg-sort').onchange=function(e){state.sort=e.target.value;render();};
 
    function clearActive(){state.selected=null;document.querySelectorAll('.dg-card.dg-active').forEach(function(c){c.classList.remove('dg-active');});}
    function setActive(i,fly){
      if(fly===undefined)fly=true;
      clearActive();state.selected=i;
      var card=document.querySelector('[data-i="'+i+'"]');
      if(card){card.classList.add('dg-active');card.scrollIntoView({block:'nearest',behavior:'smooth'});}
      if(fly)map.flyTo([PLACES[i].lat,PLACES[i].lng],16,{duration:.6});
      markers[i].openPopup();
    }
function render(){
  var items=PLACES.map(function(p,i){
      var o={};
      for(var k in p)o[k]=p[k];
      o.i=i;
      return o;
    })
    .filter(function(p){return state.active.has(p.cat);})
    .sort(function(a,b){
      if(state.sort==='rating')return b.rating-a.rating||a.dist-b.dist;
      if(state.sort==='name')return a.name.localeCompare(b.name);
      return a.dist-b.dist;
    });

  // NEW: update visible markers on the map
  var visibleMarkers = {};
  items.forEach(function(p){
    visibleMarkers[p.i] = true;
  });

  Object.keys(markers).forEach(function(i){
    if(visibleMarkers[i]){
      if(!map.hasLayer(markers[i])){
        markers[i].addTo(map);
      }
    } else {
      if(map.hasLayer(markers[i])){
        map.removeLayer(markers[i]);
      }
    }
  });

  // If selected item is no longer visible, clear selection
  if(state.selected !== null && !visibleMarkers[state.selected]){
    state.selected = null;
  }

  countEl.textContent=items.length+' place'+(items.length!==1?'s':'')+' shown';
  listEl.innerHTML='';

  items.forEach(function(p){
    var c=document.createElement('div');
    c.className='dg-card'+(state.selected===p.i?' dg-active':'');
    c.setAttribute('data-i',p.i);

    c.innerHTML=
      '<div class="dg-ctop"><div class="dg-name">'+p.name+'</div><div class="dg-rating">★ '+p.rating.toFixed(1)+'</div></div>'+
      '<div class="dg-meta"><span class="dg-tag"><span class="dg-dot" style="background:'+CATS[p.cat].color+'"></span>'+p.cat+'</span>'+
      '<span class="dg-walk">🚶 '+p.walk+' min · '+distLabel(p.dist)+'</span><span class="dg-price">'+euro(p.price)+'</span>'+
      '<span class="dg-closed">'+p.closed+'</span></div>'+
      '<a class="dg-dir" href="'+p.url+'" target="_blank" rel="noopener" onclick="event.stopPropagation()">'+
      '<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M12 21s-7-7.2-7-11a7 7 0 1114 0c0 3.8-7 11-7 11z"/><circle cx="12" cy="10" r="2.5"/></svg>Directions</a>';

    c.onclick=function(){setActive(p.i);};
    listEl.appendChild(c);
  });
}
    render();
    setTimeout(function(){map.invalidateSize();},200);
  }
  if(document.readyState!=='loading')init();else document.addEventListener('DOMContentLoaded',init);
})();
